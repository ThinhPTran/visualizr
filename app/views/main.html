Getting started

About the project

G3 is a Javascript charting library, built on top of d3.js. It's primary purpose is to allow easy access to dynamic in broswer geo-oriented charts.

G3 uses d3.js and jQuery as dependencies.

Quick Start

Finish this section when we are ready to push and have a bower and cdn version of G3 registered.

Contribute

Plots

g3.plot

Example

Plots can have a width set explicitly or will size based off of their container div.

var myPlot = g3.plot('plot_div')
  .height(400)
  .xDomain([0, 500])
  .yDomain([0, 300])
  .draw();

Logs

g3.log

Example

var myLogPlot = g3.plot('.log_plot_div')
  .height(400).xDomain([0, 300])
  .yDomain([0, dataArray.length])
  .draw();

var myLog = g3.log(myLogPlot, dataArray).draw();
  
var myLog2 = g3.log(myLogPlot, dataArray)
  .color('black')
  .xMult(0.5)
  .xTrans(200)
  .draw();

Wiggles

g3.wiggle

Example

var wigglePlot = g3.plot('.wiggle_plot_div')
  .height(400)
  .xDomain([-1000, 1000])
  .yDomain([0, twoDArray[0].length])
  .draw();

var wig = g3.wiggle(wigglePlot, twoDArray)
  .skip(50)
  .xMult(0.04)
  .sampleRate(10)
  .xTrans(-800)
  .draw();

Seismics

g3.seismic

Example

var seismicPlot = g3.plot('.seismic_plot_div')
  .height(400)
  .xDomain([0, twoDArray.length - 1])
  .yDomain([0, twoDArray[0].length - 1])
  .draw();

var max = twoDArray.max();
var colorScale = d3.scale.linear()
  .domain([-max, 0, max])
  .range(['#FF0000', '#FFF', '#0000FF']);

var seismic = g3.seismic(seismicPlot, twoDArray)
  .colorScale(colorScale)
  .draw();

reDraw function n/a canvas.reDraw(newDataArray)

Horizons

g3.horizon

Example

// Get a cross section from the 2d array
var arr = [];
for(var i = 0; i < twoDArray.length; i++){
  arr.push(twoDArray[i][0]);
}

// Grab the min and max values from the array
var max = d3.max(arr);
var min = d3.min(arr);

var horizonPlot = g3.plot('.horizon_plot_div')
.height(400)
.xDomain([0, twoDArray.length])
.yDomain([min, max])
.draw();

var horizon = g3.horizon(horizonPlot, arr).draw();
